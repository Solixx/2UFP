#include <stdio.h>
#include <pthread.h>

double fat = 0;

void* fatorial(void* arg){
	int n = *(int *) arg;
	fat = n;
	for(int i = n-1; i > 0; i--){
		fat *= i;
	}
	pthread_exit(NULL);
}

int main(){
	pthread tarefa;
	int n;

	scanf("%d", &n);
	pthread_create(&tarefa, NULL, fatorial, (void *) &n);
	pthread_join(tarefa, NULL);
	printf("Fatorial=%ld\n", fat);
	return 0;
}
